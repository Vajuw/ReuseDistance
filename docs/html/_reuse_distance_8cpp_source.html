00154 // this should be fast as possible. This code is from http://graphics.stanford.edu/~seander/bithacks.html#IntegerLog
00155 static const uint64_t b[] = {0x2L, 0xCL, 0xF0L, 0xFF00L, 0xFFFF0000L, 0xFFFFFFFF00000000L};
00156 static const uint32_t S[] = {1, 2, 4, 8, 16, 32};
00157 inline uint64_t ShaveBitsPwr2(uint64_t val){
00158     val -= 1;
00159     register uint64_t r = 0; // result of log2(v) will go here
00160     for (int32_t i = 5; i >= 0; i--){
00161         if (val & b[i]){
00162             val = val >> S[i];
00163             r |= S[i];
00164         }
00165     }
00166     return (2 << r);
