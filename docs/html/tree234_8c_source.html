00090 static int countnode234(node234 *n) {
00091     int count = 0;
00092     int i;
00093     if (!n)
00094         return 0;
00095     for (i = 0; i < 4; i++)
00096         count += n->counts[i];
00097     for (i = 0; i < 3; i++)
00098         if (n->elems[i])
00099             count++;
00100     return count;

00404  * Look up the element at a given numeric index in a 2-3-4 tree.
00405  * Returns NULL if the index is out of range.
00406  */
00407 void *index234(tree234 *t, int index) {
00408     node234 *n;
00409 
00410     if (!t->root)
00411         return NULL;                   /* tree is empty */
00412 
00413     if (index < 0 || index >= countnode234(t->root))
00414         return NULL;                   /* out of range */
00415 
00416     n = t->root;
00417     
00418     while (n) {
00419         if (index < n->counts[0])
00420             n = n->kids[0];
00421         else if (index -= n->counts[0] + 1, index < 0)
00422             return n->elems[0];
00423         else if (index < n->counts[1])
00424             n = n->kids[1];
00425         else if (index -= n->counts[1] + 1, index < 0)
00426             return n->elems[1];
00427         else if (index < n->counts[2])
00428             n = n->kids[2];
00429         else if (index -= n->counts[2] + 1, index < 0)
00430             return n->elems[2];
00431         else
00432             n = n->kids[3];
00433     }

00436     return NULL;

00439 /*
00440  * Find an element e in a sorted 2-3-4 tree t. Returns NULL if not
00441  * found. e is always passed as the first argument to cmp, so cmp
00442  * can be an asymmetric function if desired. cmp can also be passed
00443  * as NULL, in which case the compare function from the tree proper
00444  * will be used.
00445  */
00446 void *findrelpos234(tree234 *t, void *e, cmpfn234 cmp,
00447                     int relation, int *index) {
00448     node234 *n;
00449     void *ret;
00450     int c;
00451     int idx, ecount, kcount, cmpret;
00452 
00453     if (t->root == NULL)
00454         return NULL;
00455 
00456     if (cmp == NULL)
00457         cmp = t->cmp;
00458 
00459     n = t->root;
00460     /*
00461      * Attempt to find the element itself.
00462      */
00463     idx = 0;
00464     ecount = -1;
00465     /*
00466      * Prepare a fake `cmp' result if e is NULL.
00467      */
00468     cmpret = 0;
00469     if (e == NULL) {
00470         assert(relation == REL234_LT || relation == REL234_GT);
00471         if (relation == REL234_LT)
00472             cmpret = +1;               /* e is a max: always greater */
00473         else if (relation == REL234_GT)
00474             cmpret = -1;               /* e is a min: always smaller */
00475     }
00476     while (1) {
00477         for (kcount = 0; kcount < 4; kcount++) {
00478             if (kcount >= 3 || n->elems[kcount] == NULL ||
00479                 (c = cmpret ? cmpret : cmp(e, n->elems[kcount])) < 0) {
00480                 break;
00481             }
00482             if (n->kids[kcount]) idx += n->counts[kcount];
00483             if (c == 0) {
00484                 ecount = kcount;
00485                 break;
00486             }
00487             idx++;
00488         }
00489         if (ecount >= 0)
00490             break;
00491         if (n->kids[kcount])
00492             n = n->kids[kcount];
00493         else
00494             break;
00495     }
00496 
00497     if (ecount >= 0) {
00498         /*
00499          * We have found the element we're looking for. It's
00500          * n->elems[ecount], at tree index idx. If our search
00501          * relation is EQ, LE or GE we can now go home.
00502          */
00503         if (relation != REL234_LT && relation != REL234_GT) {
00504             if (index) *index = idx;
00505             return n->elems[ecount];
00506         }
00507 
00508         /*
00509          * Otherwise, we'll do an indexed lookup for the previous
00510          * or next element. (It would be perfectly possible to
00511          * implement these search types in a non-counted tree by
00512          * going back up from where we are, but far more fiddly.)
00513          */
00514         if (relation == REL234_LT)
00515             idx--;
00516         else
00517             idx++;
00518     } else {
00519         /*
00520          * We've found our way to the bottom of the tree and we
00521          * know where we would insert this node if we wanted to:
00522          * we'd put it in in place of the (empty) subtree
00523          * n->kids[kcount], and it would have index idx
00524          * 
00525          * But the actual element isn't there. So if our search
00526          * relation is EQ, we're doomed.
00527          */
00528         if (relation == REL234_EQ)
00529             return NULL;
00530 
00531         /*
00532          * Otherwise, we must do an index lookup for index idx-1
00533          * (if we're going left - LE or LT) or index idx (if we're
00534          * going right - GE or GT).
00535          */
00536         if (relation == REL234_LT || relation == REL234_LE) {
00537             idx--;
00538         }
00539     }
00540 
00541     /*
00542      * We know the index of the element we want; just call index234
00543      * to do the rest. This will return NULL if the index is out of
00544      * bounds, which is exactly what we want.
00545      */
00546     ret = index234(t, idx);
00547     if (ret && index) *index = idx;

00560 /*
00561  * Delete an element e in a 2-3-4 tree. Does not free the element,
00562  * merely removes all links to it from the tree nodes.
00563  */
00564 static void *delpos234_internal(tree234 *t, int index) {
00565     node234 *n;
00566     void *retval;
00567     int ei = -1;
00568 
00569     retval = 0;
00570 
00571     n = t->root;
00572     LOG(("deleting item %d from tree %p\n", index, t));
00573     while (1) {
00574         while (n) {
00575             int ki;
00576             node234 *sub;
00577 
00578             LOG(("  node %p: %p/%d [%p] %p/%d [%p] %p/%d [%p] %p/%d index=%d\n",
00579                  n,
00580                  n->kids[0], n->counts[0], n->elems[0],
00581                  n->kids[1], n->counts[1], n->elems[1],
00582                  n->kids[2], n->counts[2], n->elems[2],
00583                  n->kids[3], n->counts[3],
00584                  index));
00585             if (index < n->counts[0]) {
00586                 ki = 0;
00587             } else if (index -= n->counts[0]+1, index < 0) {
00588                 ei = 0; break;
00589             } else if (index < n->counts[1]) {
00590                 ki = 1;
00591             } else if (index -= n->counts[1]+1, index < 0) {
00592                 ei = 1; break;
00593             } else if (index < n->counts[2]) {
00594                 ki = 2;
00595             } else if (index -= n->counts[2]+1, index < 0) {
00596                 ei = 2; break;
00597             } else {
00598                 ki = 3;
00599             }
00600             /*
00601              * Recurse down to subtree ki. If it has only one element,
00602              * we have to do some transformation to start with.
00603              */
00604             LOG(("  moving to subtree %d\n", ki));
00605             sub = n->kids[ki];
00606             if (!sub->elems[1]) {
00607                 LOG(("  subtree has only one element!\n", ki));
00608                 if (ki > 0 && n->kids[ki-1]->elems[1]) {
00609                     /*
00610                      * Case 3a, left-handed variant. Child ki has
00611                      * only one element, but child ki-1 has two or
00612                      * more. So we need to move a subtree from ki-1
00613                      * to ki.
00614                      * 
00615                      *                . C .                     . B .
00616                      *               /     \     ->            /     \
00617                      * [more] a A b B c   d D e      [more] a A b   c C d D e
00618                      */
00619                     node234 *sib = n->kids[ki-1];
00620                     int lastelem = (sib->elems[2] ? 2 :
00621                                     sib->elems[1] ? 1 : 0);
00622                     sub->kids[2] = sub->kids[1];
00623                     sub->counts[2] = sub->counts[1];
00624                     sub->elems[1] = sub->elems[0];
00625                     sub->kids[1] = sub->kids[0];
00626                     sub->counts[1] = sub->counts[0];
00627                     sub->elems[0] = n->elems[ki-1];
00628                     sub->kids[0] = sib->kids[lastelem+1];
00629                     sub->counts[0] = sib->counts[lastelem+1];
00630                     if (sub->kids[0]) sub->kids[0]->parent = sub;
00631                     n->elems[ki-1] = sib->elems[lastelem];
00632                     sib->kids[lastelem+1] = NULL;
00633                     sib->counts[lastelem+1] = 0;
00634                     sib->elems[lastelem] = NULL;
00635                     n->counts[ki] = countnode234(sub);
00636                     LOG(("  case 3a left\n"));
00637                     LOG(("  index and left subtree count before adjustment: %d, %d\n",
00638                          index, n->counts[ki-1]));
00639                     index += n->counts[ki-1];
00640                     n->counts[ki-1] = countnode234(sib);
00641                     index -= n->counts[ki-1];
00642                     LOG(("  index and left subtree count after adjustment: %d, %d\n",
00643                          index, n->counts[ki-1]));
00644                 } else if (ki < 3 && n->kids[ki+1] &&
00645                            n->kids[ki+1]->elems[1]) {
00646                     /*
00647                      * Case 3a, right-handed variant. ki has only
00648                      * one element but ki+1 has two or more. Move a
00649                      * subtree from ki+1 to ki.
00650                      * 
00651                      *      . B .                             . C .
00652                      *     /     \                ->         /     \
00653                      *  a A b   c C d D e [more]      a A b B c   d D e [more]
00654                      */
00655                     node234 *sib = n->kids[ki+1];
00656                     int j;
00657                     sub->elems[1] = n->elems[ki];
00658                     sub->kids[2] = sib->kids[0];
00659                     sub->counts[2] = sib->counts[0];
00660                     if (sub->kids[2]) sub->kids[2]->parent = sub;
00661                     n->elems[ki] = sib->elems[0];
00662                     sib->kids[0] = sib->kids[1];
00663                     sib->counts[0] = sib->counts[1];
00664                     for (j = 0; j < 2 && sib->elems[j+1]; j++) {
00665                         sib->kids[j+1] = sib->kids[j+2];
00666                         sib->counts[j+1] = sib->counts[j+2];
00667                         sib->elems[j] = sib->elems[j+1];
00668                     }
00669                     sib->kids[j+1] = NULL;
00670                     sib->counts[j+1] = 0;
00671                     sib->elems[j] = NULL;
00672                     n->counts[ki] = countnode234(sub);
00673                     n->counts[ki+1] = countnode234(sib);
00674                     LOG(("  case 3a right\n"));
00675                 } else {
00676                     /*
00677                      * Case 3b. ki has only one element, and has no
00678                      * neighbour with more than one. So pick a
00679                      * neighbour and merge it with ki, taking an
00680                      * element down from n to go in the middle.
00681                      *
00682                      *      . B .                .
00683                      *     /     \     ->        |
00684                      *  a A b   c C d      a A b B c C d
00685                      * 
00686                      * (Since at all points we have avoided
00687                      * descending to a node with only one element,
00688                      * we can be sure that n is not reduced to
00689                      * nothingness by this move, _unless_ it was
00690                      * the very first node, ie the root of the
00691                      * tree. In that case we remove the now-empty
00692                      * root and replace it with its single large
00693                      * child as shown.)
00694                      */
00695                     node234 *sib;
00696                     int j;
00697 
00698                     if (ki > 0) {
00699                         ki--;
00700                         index += n->counts[ki] + 1;
00701                     }
00702                     sib = n->kids[ki];
00703                     sub = n->kids[ki+1];
00704 
00705                     sub->kids[3] = sub->kids[1];
00706                     sub->counts[3] = sub->counts[1];
00707                     sub->elems[2] = sub->elems[0];
00708                     sub->kids[2] = sub->kids[0];
00709                     sub->counts[2] = sub->counts[0];
00710                     sub->elems[1] = n->elems[ki];
00711                     sub->kids[1] = sib->kids[1];
00712                     sub->counts[1] = sib->counts[1];
00713                     if (sub->kids[1]) sub->kids[1]->parent = sub;
00714                     sub->elems[0] = sib->elems[0];
00715                     sub->kids[0] = sib->kids[0];
00716                     sub->counts[0] = sib->counts[0];
00717                     if (sub->kids[0]) sub->kids[0]->parent = sub;
00718 
00719                     n->counts[ki+1] = countnode234(sub);
00720 
00721                     sfree(sib);
00722 
00723                     /*
00724                      * That's built the big node in sub. Now we
00725                      * need to remove the reference to sib in n.
00726                      */
00727                     for (j = ki; j < 3 && n->kids[j+1]; j++) {
00728                         n->kids[j] = n->kids[j+1];
00729                         n->counts[j] = n->counts[j+1];
00730                         n->elems[j] = j<2 ? n->elems[j+1] : NULL;
00731                     }
00732                     n->kids[j] = NULL;
00733                     n->counts[j] = 0;
00734                     if (j < 3) n->elems[j] = NULL;
00735                     LOG(("  case 3b ki=%d\n", ki));
00736 
00737                     if (!n->elems[0]) {
00738                         /*
00739                          * The root is empty and needs to be
00740                          * removed.
00741                          */
00742                         LOG(("  shifting root!\n"));
00743                         t->root = sub;
00744                         sub->parent = NULL;
00745                         sfree(n);
00746                     }
00747                 }
00748             }
00749             n = sub;
00750         }
00751         if (!retval)
00752             retval = n->elems[ei];
00753 
00754         if (ei==-1)
00755             return NULL;               /* although this shouldn't happen */
00756 
00757         /*
00758          * Treat special case: this is the one remaining item in
00759          * the tree. n is the tree root (no parent), has one
00760          * element (no elems[1]), and has no kids (no kids[0]).
00761          */
00762         if (!n->parent && !n->elems[1] && !n->kids[0]) {
00763             LOG(("  removed last element in tree\n"));
00764             sfree(n);
00765             t->root = NULL;
00766             return retval;
00767         }
00768 
00769         /*
00770          * Now we have the element we want, as n->elems[ei], and we
00771          * have also arranged for that element not to be the only
00772          * one in its node. So...
00773          */
00774 
00775         if (!n->kids[0] && n->elems[1]) {
00776             /*
00777              * Case 1. n is a leaf node with more than one element,
00778              * so it's _really easy_. Just delete the thing and
00779              * we're done.
00780              */
00781             int i;
00782             LOG(("  case 1\n"));
00783             for (i = ei; i < 2 && n->elems[i+1]; i++)
00784                 n->elems[i] = n->elems[i+1];
00785             n->elems[i] = NULL;
00786             /*
00787              * Having done that to the leaf node, we now go back up
00788              * the tree fixing the counts.
00789              */
00790             while (n->parent) {
00791                 int childnum;
00792                 childnum = (n->parent->kids[0] == n ? 0 :
00793                             n->parent->kids[1] == n ? 1 :
00794                             n->parent->kids[2] == n ? 2 : 3);
00795                 n->parent->counts[childnum]--;
00796                 n = n->parent;
00797             }
00798             return retval;             /* finished! */
00799         } else if (n->kids[ei]->elems[1]) {
00800             /*
00801              * Case 2a. n is an internal node, and the root of the
00802              * subtree to the left of e has more than one element.
00803              * So find the predecessor p to e (ie the largest node
00804              * in that subtree), place it where e currently is, and
00805              * then start the deletion process over again on the
00806              * subtree with p as target.
00807              */
00808             node234 *m = n->kids[ei];
00809             void *target;
00810             LOG(("  case 2a\n"));
00811             while (m->kids[0]) {
00812                 m = (m->kids[3] ? m->kids[3] :
00813                      m->kids[2] ? m->kids[2] :
00814                      m->kids[1] ? m->kids[1] : m->kids[0]);                  
00815             }
00816             target = (m->elems[2] ? m->elems[2] :
00817                       m->elems[1] ? m->elems[1] : m->elems[0]);
00818             n->elems[ei] = target;
00819             index = n->counts[ei]-1;
00820             n = n->kids[ei];
00821         } else if (n->kids[ei+1]->elems[1]) {
00822             /*
00823              * Case 2b, symmetric to 2a but s/left/right/ and
00824              * s/predecessor/successor/. (And s/largest/smallest/).
00825              */
00826             node234 *m = n->kids[ei+1];
00827             void *target;
00828             LOG(("  case 2b\n"));
00829             while (m->kids[0]) {
00830                 m = m->kids[0];
00831             }
00832             target = m->elems[0];
00833             n->elems[ei] = target;
00834             n = n->kids[ei+1];
00835             index = 0;
00836         } else {
00837             /*
00838              * Case 2c. n is an internal node, and the subtrees to
00839              * the left and right of e both have only one element.
00840              * So combine the two subnodes into a single big node
00841              * with their own elements on the left and right and e
00842              * in the middle, then restart the deletion process on
00843              * that subtree, with e still as target.
00844              */
00845             node234 *a = n->kids[ei], *b = n->kids[ei+1];
00846             int j;
00847 
00848             LOG(("  case 2c\n"));
00849             a->elems[1] = n->elems[ei];
00850             a->kids[2] = b->kids[0];
00851             a->counts[2] = b->counts[0];
00852             if (a->kids[2]) a->kids[2]->parent = a;
00853             a->elems[2] = b->elems[0];
00854             a->kids[3] = b->kids[1];
00855             a->counts[3] = b->counts[1];
00856             if (a->kids[3]) a->kids[3]->parent = a;
00857             sfree(b);
00858             n->counts[ei] = countnode234(a);
00859             /*
00860              * That's built the big node in a, and destroyed b. Now
00861              * remove the reference to b (and e) in n.
00862              */
00863             for (j = ei; j < 2 && n->elems[j+1]; j++) {
00864                 n->elems[j] = n->elems[j+1];
00865                 n->kids[j+1] = n->kids[j+2];
00866                 n->counts[j+1] = n->counts[j+2];
00867             }
00868             n->elems[j] = NULL;
00869             n->kids[j+1] = NULL;
00870             n->counts[j+1] = 0;
00871             /*
00872              * It's possible, in this case, that we've just removed
00873              * the only element in the root of the tree. If so,
00874              * shift the root.
00875              */
00876             if (n->elems[0] == NULL) {
00877                 LOG(("  shifting root!\n"));
00878                 t->root = a;
00879                 a->parent = NULL;
00880                 sfree(n);
00881             }
00882             /*
00883              * Now go round the deletion process again, with n
00884              * pointing at the new big node and e still the same.
00885              */
00886             n = a;
00887             index = a->counts[0] + a->counts[1] + 1;
00888         }
00889     }
00890 }

00963     /* Count the non-NULL kids. */

00966     for (i = nkids; i < 4; i++)

00977     /* Ensure no elements beyond the first NULL are non-NULL. */
00978     for (i = nelems; i < 3; i++)

00980             error("node %p: nelems=%d but elems[%d] non-NULL",
00981                    node, nelems, i);
00982         }

00987          * depth is the same everywhere.
00988          */

00990             ctx->treedepth = level;    /* we didn't know the depth yet */
00991         else if (ctx->treedepth != level)
00992             error("node %p: leaf at depth %d, previously seen depth %d",
00993                    node, level, ctx->treedepth);

00995         /*
00996          * If nkids != 0, then it should be nelems+1, unless nelems
00997          * is 0 in which case nkids should also be 0 (and so we
00998          * shouldn't be in this condition at all).
00999          */
01000         int shouldkids = (nelems ? nelems+1 : 0);
01001         if (nkids != shouldkids) {
01002             error("node %p: %d elems should mean %d kids but has %d",
01003                    node, nelems, shouldkids, nkids);
01004         }
01005     }
01006 
01007     /*
01008      * nelems should be at least 1.
01009      */
01010     if (nelems == 0) {
01011         error("node %p: no elems", node, nkids);
01012     }

01015      * Add nelems to the running element count of the whole tree.
01016      */
01017     ctx->elemcount += nelems;
01018 
01019     /*
01020      * Check ordering property: all elements should be strictly >
01021      * lowbound, strictly < highbound, and strictly < each other in
01022      * sequence. (lowbound and highbound are NULL at edges of tree
01023      * - both NULL at root node - and NULL is considered to be <
01024      * everything and > everything. IYSWIM.)
01025      */
01026     if (cmp) {
01027         for (i = -1; i < nelems; i++) {
01028             void *lower = (i == -1 ? lowbound : node->elems[i]);
01029             void *higher = (i+1 == nelems ? highbound : node->elems[i+1]);
01030             if (lower && higher && cmp(lower, higher) >= 0) {
01031                 error("node %p: kid comparison [%d=%s,%d=%s] failed",
01032                       node, i, lower, i+1, higher);
01033             }
01034         }
01035     }
01036 
01037     /*

01039      * parent pointer coming back to this node.
01040      */
01041     for (i = 0; i < nkids; i++)
01042         if (node->kids[i]->parent != node) {
01043             error("node %p kid %d: parent ptr is %p not %p",
01044                    node, i, node->kids[i]->parent, node);
01045         }
01046 
01047 
01048     /*
01049      * Now (finally!) recurse into subtrees.
01050      */
01051     count = nelems;
01052 
01053     for (i = 0; i < nkids; i++) {
01054         void *lower = (i == 0 ? lowbound : node->elems[i-1]);
01055         void *higher = (i >= nelems ? highbound : node->elems[i]);
01056         int subcount = chknode(ctx, level+1, node->kids[i], lower, higher);
01057         if (node->counts[i] != subcount) {
01058             error("node %p kid %d: count says %d, subtree really has %d",
01059                   node, i, node->counts[i], subcount);
01060         }
01061         count += subcount;
01062     }
01063 
01064     return count;
